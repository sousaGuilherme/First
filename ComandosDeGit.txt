Comandos De GIT

// inicia o git (repositório) no seu projeto
git init

// verificar alterações de pastas e arquivos no projeto
git status

// adiciona todos os arquivos e pastas modificados, ao stage area
git add .

// cria e descreve um ponto na história
git comit -m "message here"

// histórico de commits do projeto
git log

// para remover do stage area
git rm --cached file

// restaurar do ultimo save
git restore file

// para voltar em alguns saves do seu projeto
git checkout "codigo da log"

// puxa modificações do repositório remoto para o repositório local
git pull

// envia as modificações locais para o repositório remoto
git push

//trazendo um repositório remoto para minha maquina
git clone LinkDoGitHub
